LARAVEL PROJECT GUIDELINES
1- Create our Laravel project.
2- Install the breeze.
3- We set the authentication for breeze. => AuthenticatedSessionController.php file
4- We set the seeder file admin and user.
5- To do the CRUD operation, we need three files, index, create and edit.
6- In Route file, we need the following routes, home, create, store, edit, update and delete.
Note: In edit and delete routes, we need to pass the id.
7- In Controller file, again we need the following functions, home, create, store, edit, update and delete.

3- studentcontroller file
<?php

namespace App\Http\Controllers;
use App\Models\Student;
use Illuminate\Http\Request;

class StudentController extends Controller
{
    public function student(){
        return view('student');
    }


    public function store_student(Request $request){

    $validated = $request->validate([
        'name' => 'required|string|max:20',
        'last_name' => 'required|string|max:20',
        'class' => 'required|string|max:20',
        'email' => 'required|email|unique:students,email',
    ]);

    Student::create([
        'name' => $validated['name'],
        'last_name' => $validated['last_name'],
        'class'     => $validated['class'],
        'email'     => $validated['email'],
    ]);

    return redirect()->route('student')->with('success', 'Student inserted successfully!');
}

}

student routes
Route::get('/student', [StudentController::class, 'student'])->name('student');
Route::post('/store-student', [StudentController::class, 'store_student'])->name('store-student');
